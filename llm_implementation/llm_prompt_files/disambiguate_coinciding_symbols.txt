{
    "conflict_1": {

    },

        "parent": "<msubsup id="mathml_msubsup_id_37"><mi id="mathml_mi_id_59">b</mi><mi id="mathml_mi_id_3">o</mi><mi id="mathml_mi_id_43">d</mi></msubsup>",
        "child": "<mi id="mathml_mi_id_43">d</mi>"
    },
    "conflict_3":{
        "parent": "<mi id="mathml_mi_id_5">P</mi><mi id="mathml_mi_id_78">A</mi>",
        "child": "<mi id="mathml_mi_id_5">P</mi>"
